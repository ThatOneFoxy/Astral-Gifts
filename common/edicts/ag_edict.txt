ag_crisis_cont = {
    icon = "GFX_edict_type_policy"
    length = 1

    potential = {
        AND = {
            has_technology = tech_crisis_edicts
            has_country_flag = TR_host
            NOT = {
                has_country_flag = locked_crisis
            }
        }
    }
    resources = { category = edicts cost = { energy = 1 } }
    effect = {
        country_event = { id = crisis.2005 }
    }
}
