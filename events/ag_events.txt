            chance = 5

                limit = {
                    NOT = {
                        has_country_flag = techharvestingunlocked

                }

                tech = tech_astral_harvesting
                progress = 0.25
            }
            set_country_flag = techharvestingunlocked

                add_resource = { astral_threads = 150 }

            chance = 10

        random = {
            chance = 15
            add_resource = { astral_threads = 50 }
        }
        random = {
            chance = 70

        }
    }
}
